8.2 Functions as first class object

def sumofsqaures(a, b, c):
    return square(a) + square(b) + square(c)

def sumofcubes(a, b, c):
    return cube(a) + cube(b) + cube(c)

def square(x):
    return x*x

def cube(x):
    return x*x*x

def sumof(a, b, c, func):
    return func(a) + func(b) + func(c)
